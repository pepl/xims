#
# The following commands will convert an existing PostgreSQL database with
# latin1 encoding to a new database with unicode encoding.
#
# You should execute them as the postgresql user (postgres, pgsql)
#
# Read carefully, think, and use them at your own risk!
#

pg_dump xims > dump_latin1.sql
iconv -f iso8859-1 -t utf-8 < dump_latin1.sql > dump-utf-8.sql
createdb --encoding=unicode xims_unicode
psql xims_unicode < dump-utf-8.sql
